import org.camunda.bpm.engine.delegate.JavaDelegate;

 * Carries the annotation in the explicit normal form {@code value = {...}}

 * pair by name.
 */

public class NormalAnnotatedDelegate implements JavaDelegate {

    @Override
    public void execute(DelegateExecution execution) {
        // annotated in normal form
    }
}
